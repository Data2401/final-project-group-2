The Data Collaboration

The “Cause of Deaths” dataset provides the counts, but the “World Population” dataset provides the context.

By merging them, we move from raw totals to meaningful rates. This allows us to answer not just “how many died?”, but “how heavily was the population actually impacted?”

regional_summary %>%
  group_by(Continent) %>%
  summarise(Avg_Regional_Burden = mean(Avg_Rate, na.rm = TRUE), .groups = "drop") %>%
  ggplot(aes(x = reorder(Continent, Avg_Regional_Burden), y = Avg_Regional_Burden, fill = Continent)) +
  geom_col() +
  coord_flip() +
  theme_minimal() +
  labs(title = "Average Disease Burden by Region",
       x = "Continent",
       y = "Average Death Rate") +
  guides(fill = "none")

ntrl_deaths %>%
  filter(Year %in% c(1990, 2000, 2010, 2015)) %>%
  pivot_longer(
    cols = all_of(disease_cols),
    names_to = "Disease",
    values_to = "Deaths"
  ) %>%
  filter(Deaths > 100) %>%  # removes tiny irrelevant values
  group_by(Disease) %>%
  summarise(Countries.Affected = n_distinct(Country.Territory), .groups = "drop") %>%
  arrange(desc(Countries.Affected)) %>%
  slice_head(n = 10) %>%
  ggplot(aes(x = reorder(Disease, Countries.Affected), y = Countries.Affected)) +
  geom_col(fill = "steelblue") +
  geom_text(aes(label = Countries.Affected), hjust = -0.2, size = 5) +
  coord_flip() +
  theme_minimal() +
  labs(title = "Most Widespread Diseases (Global Reach)",
       subtitle = "Top 8 diseases by number of countries affected",
       x = "Disease",
       y = "Countries Affected") +
  ylim(0, NA)
