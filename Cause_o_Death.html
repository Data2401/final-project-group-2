Death Rates Over Time Based in Continents

If we compare every continents death rates over the year, we may see africa has the highest percentages compared to other continents, but thats not all we see.

disease_country_count <- death_ds %>%
  pivot_longer(
    cols = -c(Country.Territory, Code, Year, Total.Death),
    names_to = "Disease",
    values_to = "Deaths"
  ) %>%
  filter(Deaths > 0) %>%
  group_by(Disease) %>%
  summarize(
    Countries.Affected = n_distinct(Country.Territory),
    .groups = "drop"
  ) %>%
  arrange(desc(Countries.Affected))

disease_totals <- death_ds %>%
  pivot_longer(
    cols = -c(Country.Territory, Code, Year, Total.Death),
    names_to = "Disease",
    values_to = "Deaths"
  ) %>%
  group_by(Disease) %>%
  summarize(
    Total.Deaths = sum(Deaths, na.rm = TRUE),
